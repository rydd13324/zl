end

    if code==200 and content then

      总表={}

      卡片=截取:gmatch('three">(.-)</li>')

      for i in 卡片 do

        table.insert(总表,i)

      end


      for k,v in ipairs(总表)

        标题=v:match('">(.-)</a>')
        链接=v:match('<a href="(.-)">')

            bt=标题,

          }



        end

      end
    end
    刷新.getRefreshFooter().setAnimatingColor(0xFF73ABFE)
  end)

end
